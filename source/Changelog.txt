	UPDATE=5

	OUTPUT=D:/mywebserverdir/serverstatus.html

	{
	TYPE=PLAYER

	INPUT=C:/uox3/dfndata/html/player.htf
	OUTPUT=D:/mywebserverdir/player.html
	}

	[DEFAULT_GUILD]

	TYPE=GUILD

	}
